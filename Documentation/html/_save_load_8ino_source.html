117 bool SaveLoadClass::SaveCfg (int page) {
118  uint16_t i;
119  int MemPointer;
120 
121  // Load address of selected page
122  if (page == -1) page = CurrentPage; // No value = Current Bank
123  if (page < 0 || page > MAXSAVEPAGES - 1) return false;
124 
125  // Write data
126  MemPointer = PageBase[page];
127  // Mark the page as containing Cfg data
128  EEPROM.put (MemPointer, (uint16_t)CFGDATATAG);
129  MemPointer += sizeof (uint16_t);
130  // Reserved PAGEGENERALeeSize for general data
131  MemPointer += PAGEGENERALeeSize-(MemPointer-PageBase[page]);// sizeof (uint16_t);
132 
133  // Bank data
134  for (i = 0; i < 9; i++) {
135  MemPointer += CVControls[i].SaveCfg (MemPointer);
136  }
137 
138  #ifdef PRINTDEBUG
139  char outBuf[80];
140  sprintf(outBuf, "Saved: %1d/%4lu/%4d", page, PageBase[page], MemPointer);
141  Serial.println( outBuf);
142  #endif
143  #ifdef DEBUGMODE
144  char outBuf2[80];
145  sprintf(outBuf, "Saved: %1d/%4lu/%4d", page, PageBase[page], MemPointer);
146  SetMessageText (outBuf2);
147  #endif
148 
149  CurrentPage= page;
150  theApp.theGlobalCfg.SetPage(page);
151 
152  return true;
153 }

161 bool SaveLoadClass::LoadCfg (int page) {
162  int i;
163  uint16_t CfgDataCheck;
164  int MemPointer;
165 
166  if (page == -1) page = CurrentPage; // No value = Current Bank
167  // Load address of selected page
168  if (page < 0 || page > MAXSAVEPAGES - 1) return false;
169  // Load data index
170  MemPointer = PageBase[page];
171  // Check if the page contains Cfg data
172  EEPROM.get (MemPointer, CfgDataCheck);
173  if (CfgDataCheck != CFGDATATAG) {
174  myMenu.setupPopup ("Incorrect config data", 5000, 0, 17);
175  return false;
176  }
177  // Reserved PAGEGENERALeeSize for general data
178  MemPointer += PAGEGENERALeeSize-(MemPointer-PageBase[page]);// sizeof (uint16_t);
179 
180  // Bank data
181  for (i = 0; i < 9; i++) {
182  MemPointer += CVControls[i].LoadCfg (MemPointer);
183  }
184  CurrentPage= page;
185  theApp.theGlobalCfg.SetPage(page);
186 
187 #ifdef DEBUGMODE
188  char outBuf[80];
189  sprintf(outBuf, "Load: %1d/%4lu/%4d", page, PageBase[page], MemPointer);
190  SetMessageText (outBuf);
191 #endif
192  return true;
193 }

202 bool SaveLoadClass::LoadSysExCfg (int page) {
203  uint16_t *CfgDataCheck;
204  int MemPointer;
205 
206  if (page == -1) page = CurrentPage; // No value = Current Bank
207  // Load address of selected page
208  if (page < 0 || page > MAXSAVEPAGES - 1) return false;
209  // Decode and get CheckSum
210  unsigned csum = calcCSum ();
211  if (!csum || csum != msgCsum) {
212  myMenu.setupPopup ("No valid SysEx", 5000, 0, 17);
213  return false; // CheckSum not OK
214  }
215  // Check Cfg data
216  CfgDataCheck = (uint16_t *)rcvData;
217  if (*CfgDataCheck != CFGDATATAG) {
218  return false;
219  }
220  // Load data in EEPROM
221  MemPointer = PageBase[page];
222  EEPROM.put (MemPointer, rcvData);
223  // Load from EEPROM
224  return LoadCfg (page);
225 }
