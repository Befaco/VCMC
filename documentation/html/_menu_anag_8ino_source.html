193 bool editUserNames()
194 {
195  static uint8_t idx = 0;
196 
197  if (SetValState == 0){
198  idx= myMenu.getItemStatus();
199  if(idx<200 || idx>200+NUMUSERNAMES)
200  return true; // Cancel: no User Name selected
201  GetPortCfg()->textPort = idx;
202  idx -= 200;
203  SetValState = idx+1;
204  }
205  if (SetValState > 0)
206  {
207  if (!theApp.vMenu.EncoderselDigitStr("Name ", theApp.theGlobalCfg.UserNames[idx], SIZEPORTNAMES, 0, 45))
208  return false; // Editing name
209  SetValState = 0;
210  }
211 
212  return SelectNameMenu();
213 }
