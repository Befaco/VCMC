351 {

353  bool ret = EncoderchangeValue("Save Page", val, 1, MAXSAVEPAGES, 3, 0, 45);
354  FlashAccess->CurrentPage = val - 1;
355  if (!ret)
356  return false;
357  if (!FlashAccess->SetCurrentPage(val))

359  SetMessageText("Error Save");
