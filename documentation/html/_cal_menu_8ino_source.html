82 int updateCalSelection (int maxval) {
83  int lineMove = 0;
84 
85  // Read encoder
86  valueEncoder -= encoder->getValue ();
87 
88  // No Change in encoder
89  if (valueEncoder == lastEncoder) return 0;
90 
91  if (valueEncoder > lastEncoder) {
92  lineMove = 1;
93  PointSelected = (PointSelected < maxval) ? PointSelected + 1 : PointSelected;//0;
94  } else {
95  lineMove = -1;
96  PointSelected = (PointSelected > 0) ? PointSelected - 1 : 0;//maxval;
97  }
98 
99  lastEncoder = valueEncoder;
100  return lineMove;
101 }

111 boolean InitCalibration (void) {
112  byte UsePort = PortSelected;
113 
114  runFunc = NULL;
115 
116  // For Matrix Calibration use CV
117  if (calMode == MatrixCalMode)
118  UsePort = 2;
119  else if (UsePort < 2 && calMode!=SetClockMode)
120  return false; // return false if gate selected
121 
122  switch (UsePort) {
123  case 1:
124  PortInUse = (DemuxAnalogPort *)&CVControls[BankSelected].GateBut;
125  break;
126  case 2:
127  PortInUse = &CVControls[BankSelected].CVPort;
128  break;
129  case 3:
130  PortInUse = &CVControls[BankSelected].Slider;
131  break;
132  }
133  if( UsePort != 1){
134  oldBipolar = PortInUse->PortCfg.getInputRange();
135  rangeData = &PortInUse->PortCfg.Ranges;
136  rangeData->getDAC (OldminD, OldRangeD); // Store old DAC values
137  OldClipLow = PortInUse->PortCfg.ClipLow;
138  OldClipHigh = PortInUse->PortCfg.ClipHigh;
139  rangeData->getMIDI(OldminMIDI,OldrangeMIDI);
140  }
141  if( calMode==TwoPointCalMode||calMode == MatrixCalMode){
142 // int16_t minD, RangeD;
143 // rangeData->getMIDI(minD,RangeD);
144 // if( minD!=0 || RangeD!=120) return false; // Calibration only with 0-120 MIDI range
145  OldMIDIfunction= PortInUse->PortCfg.MIDIfunction;
146  PortInUse->PortCfg.SetMIDIFunc(PITCHTRIG); // Select Pitch trigger for calibration
147  }
148 
149 
150  PointSelected = 1;
151  PortData = -1;
152  noteRec = -1;
153  LastpointRec = 0;
154  if (calMode == MatrixCalMode) {
155  PortInUse->PortCfg.setInputRange(NOOFFSET);
156  rangeData->SetDAC (INITMINDAC, INITRANGEDAC);
157  for (int i = 0; i < 20; i++) {
158  calPoints[i] = 0;//DACPoints[i]; // Store original points (for cancel purpose)
159  pointChanged[i] = false;
160  }
161  }
162  InitCal = true;
163 
164  return true;
165 }

276  pointChanged[PointSelected] = true;
277  PointSelected = (PointSelected < 19) ? PointSelected + 1 : 0; // Step to next point
278  }
279 
280  return false;
281 }

294  /*

296  if (DACpoint < DACPoints[i] + RangeAutoCalPoint && DACpoint > DACPoints[i] - RangeAutoCalPoint)
297  return i;
298  */
299  }
300  return 0;

395 boolean TwoPointsCal () {
396  calMode = TwoPointCalMode;
397 
398  // Init Calibration
399  if (!InitCal)
400  if (!InitCalibration ()) {
401  myMenu.setupPopup ("Error:Set Note Mode", 2000, 0, 17);
402  return true; // If Init fails, abort
403  }
404  // Get the 2 calibration points
405  int32_t minD, rangeD;
406  rangeData->getDAC(minD, rangeD);
407  // bool Is5V = (rangeD > 0) ? rangeD < INITRANGEDAC / 2 + 100 : rangeD > INITRANGEDAC / 2 - 100;
408  bool Is5V = (PortInUse->PortCfg.getInputRange() == ZEROTO5V);
409 
410 
411  if( Is5V){
412  DAC1 = PortInUse->PortCfg.Ranges.Convert (12);//24);
413  DAC2 = PortInUse->PortCfg.Ranges.Convert (48);//96);
414  } else{
415  DAC1 = PortInUse->PortCfg.Ranges.Convert (24);
416  DAC2 = PortInUse->PortCfg.Ranges.Convert (96);
417  }
418 
419  // Read Port
420  //PortInUse->ReadPort (PortData);
421  PortData = PortInUse->PortValue;
422  noteRec = PortInUse->MIDIData;
423 
424  updateCalSelection (MAXMENU2POINTS); // Update Selection
425  displayTwoPoints (); // Display values
426 
427  // If button pressed (LOW input), finish
428  if (!selectionCalMode ()) {
429  uint8_t pointRec = IsCalPoint (PortData);
430  if (!AutoCal || pointRec == 0) return false;
431  if (pointRec == LastpointRec) return false;
432  LastpointRec = PointSelected = pointRec;
433  }
434  if (PointSelected < 2) { // Cancel or END Selected
435  // TODO Check here if OK or Cancel
436  ClearCalArea (); // Clear area used
437  PortInUse->PortCfg.SetMIDIFunc(OldMIDIfunction);
438  if (PointSelected == 0) // Cancel calibration, revert to previous values
439  rangeData->SetDAC (OldminD, OldRangeD); // Restore old DAC values
440  InitCal = false;
441  calMode = NoCalMode;
442  return true;
443  } else { // Accept input in point selected
444  //int32_t minD, rangeD;
445  if (PointSelected == 2) DAC1 = PortData;
446  if (PointSelected == 3) DAC2 = PortData;
447  if( Is5V){
448  minD = DAC1 - (DAC2 - DAC1) * 1.0 / 3; //2.0 / 6;
449  rangeD = (DAC2 - DAC1) * 5.0 / 3;//6;
450  } else {
451  minD = DAC1 - (DAC2 - DAC1) * 2.0 / 6.0;
452  rangeD = (DAC2 - DAC1) * 10.0 / 6.0;
453  }
454  rangeData->SetDAC (minD, rangeD);
455  }
456 
457  return false;
458 }
