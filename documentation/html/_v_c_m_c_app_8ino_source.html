53 void servicePorts () {
54 
55  if (servicingPorts) return;
56  servicingPorts = true;
57 
58  CVControls[AnagPortService].CVPort.SetDemux (); // Prepare demux for next reading
59  // Read all the gates
60  #ifndef CVTHING
61  for (int i = 0; i < NUMCHAN-1; i++) {
62  CVControls[i].ReadPorts (true, false, false);
63  CVControls[i].Blink ();
64  }
65  // Read Aux
66  if( CVControls[8].CVPort.PortCfg.IsDigitalFunc() || countCycles==3)
67  CVControls[8].ReadPorts (false, true, false);
68  if( CVControls[8].Slider.PortCfg.IsDigitalFunc() || countCycles==6)
69  CVControls[8].ReadPorts (false, false, true);
70  #endif
71  // Read CV and Faders
72  if( countCycles==0){
73  #ifndef CVTHING
74  adc->analogRead (PIN_A13);
75  adc->analogRead (PIN_A13);
76  CVControls[AnagPortService].ReadPorts (false, false, true);
77  #endif
78  } else if(countCycles!=3 && countCycles!=6)
79  CVControls[AnagPortService].ReadPorts (false, true, false);
80 
81  if( AnagPortService==7) countCycles++;
82  if( countCycles>8) countCycles=0;
83 
84  AnagPortService++;
85  if (AnagPortService >= NUMCHAN - 1) AnagPortService = 0;
86 
87  CVControls[AnagPortService].CVPort.SetDemux (); // Prepare demux for next reading
88 
89  servicingPorts = false;
90 }
