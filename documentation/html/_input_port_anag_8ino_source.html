98 bool AnalogPort::ReadPort (long &NewData) {
99  /*#ifdef DEBUGMODE
100  //if(PortNumber==PINCVINPUT)
101  beginDebugTimers ();
102  #endif*/
103  // Process MIDI clock
104  if (PortCfg.MIDIfunction == ANAGCLOCK && IntervalMIDIClock && ClockCounter) {
105  ProcessClock();
106  }
107 
108  int readMidi=MIDIData;
109  // Set Bipolar input
110  if (PortCfg.getInputRange()==MINUSPLUS5V)
111  digitalWrite (PINOFFSET, HIGH);
112  else
113  digitalWrite (PINOFFSET, LOW);
114  // Read input
115  /* uint16_t reading = 0; // Test CV inputs
116  if(PortCfg.ControllerNumber<21) reading = myMenu.disptimer / 50 / PortCfg.ControllerNumber;
117  if(PortCfg.ControllerNumber==21) reading = ((myMenu.disptimer%5000000)<500000)?0:4095<<4;
118  */
119  uint16_t reading = (uint16_t)adc->analogRead (PortNumber);//,adcNum);
120 
121  if (ANBITS > 12) reading >>= 4; // Filter first 4 bits
122  if( PortCfg.IsDigitalFunc())
123  {
124  //unsigned reading = digitalRead (PortNumber);
125  int32_t mindac,rangedac;
126  PortCfg.Ranges.getDAC(mindac,rangedac);
127  // Read as digital input
128  bool readGate = GateStatus;
129  if( reading < 4095*2/3-100){ //mindac-rangedac/3
130  reading= PortCfg.Ranges.getMaxDAC();
131  readGate = true;
132  //1readMidi = 1;
133  } else if(reading > 4095*2/3+100){
134  reading = mindac;
135  readGate = false;
136  }
137 
138  int newGate = 2;
139  if (debounceGate) { // Debounce function if selected
140  // check to see if you just pressed the button
141  // (i.e. the input went from LOW to HIGH), and you've waited
142  // long enough since the last press to ignore any noise:
143  // If the switch changed, due to noise or pressing:
144  if (readGate != lastButtonState) {
145  // reset the debouncing timer
146  lastDebounceTime = millis ();
147  }
148  if ((millis () - lastDebounceTime) > PortCfg.DelayGate) {
149  // whatever the reading is at, it's been there for longer
150  // than the debounce delay, so take it as the actual current state:
151  newGate = readGate;
152  }
153  lastButtonState = readGate;
154  } else
155  newGate = readGate;
156  if (newGate == 2) return false; // No lecture (debouncing)
157 
158 
159  NewData= reading;
160  if (reading != PortValue) { // Gate changed
161  PortValue = reading;
162  // Latch management: switch gate with edge up
163  if (PortCfg.MIDIfunction == ANAGCCLATCH || PortCfg.MIDIfunction == ANAGLATCH) {
164  if (newGate) {
165  GateStatus = !GateStatus;
166  newGate = GateStatus;
167  } else
168  return false;
169  } else
170  GateStatus = newGate;
171 
172  readMidi = (newGate)?PortCfg.Ranges.getMaxMIDI():0;
173  }
174  // NewData = PortValue = reading;
175  } else
176  {
177  // Use filter for the input
178  #ifdef FILTEREDANAGINPUT
179  FilteredInput->update (reading);
180  reading = FilteredInput->getValue ();
181  #endif
182  if( reading!=PortValue){
183  // Convert 12 bit input to MIDI values
184  readMidi = PortCfg.Ranges.invConvert (reading);
185  }
186  NewData = PortValue = reading;
187  }
188 
189  /*#ifdef DEBUGMODE
190  //if(PortNumber==PINCVINPUT)
191  endDebugTimers ();
192  #endif*/
193  if (readMidi != MIDIData) {
194  // Store new value
195  MIDIData = readMidi;
196  // Check if clock Received
197  if (PortCfg.MIDIfunction == ANAGCLOCK)
198  if( IsHigh()) ClockReceived();
199  return true;
200  }
201 
202  // Check Auto off for Note in Dude mode
203  if( msecLastMIDISent && PortCfg.AutoOff &&
204  millis()-msecLastMIDISent > PortCfg.AutoOff && PortCfg.MIDIfunction==PITCH){
205  // Note off previous note
206  MidiMerge.sendNoteOff( LastSentMIDIData, 0, PortCfg.MIDIChannel);
207  LastSentMIDIData = -999;
208  msecLastMIDISent = 0;
209  return false;
210  }
211  return false;
212 }

251 void AnalogPort::SendMIDI (int MidiData, bool GateStat) {
252  int SendData;
253 
254  if (MidiData == -9999)
255  SendData = TrimValue (MIDIData); // Use internal value
256  else
257  SendData = TrimValue (MidiData); // Use passed value
258 
259  if (SendData == LastSentMIDIData) return; // Do not send the same data twice
260 
261  switch (PortCfg.MIDIfunction) {
262  case PITCHTRIG:
263  case PITCHLEVEL:
264  case PITCH8TRIG:
265  return;
266  case PITCH:
267  // Note off previous note
268  //MidiMerge.sendNoteOff( MidiMerge.PitchData[PortCfg.MIDIChannel-1], 0, PortCfg.MIDIChannel);
269  MidiMerge.sendNoteOff( LastSentMIDIData, 0, PortCfg.MIDIChannel);
270  MidiMerge.PitchData[PortCfg.MIDIChannel-1] = SendData;
271  // Send new Note
272  MidiMerge.sendNoteOn( SendData, MidiMerge.VelData[PortCfg.MIDIChannel-1], PortCfg.MIDIChannel);
273  break;
274  case VELOCITY:
275  MidiMerge.VelData[PortCfg.MIDIChannel - 1] = SendData;
276  break; // do not send MIDI, gate button or pitch change will trigger sending data
277  case CC14BITS:
278  if(PortCfg.ControllerNumber>32)
279  break; // Only for CC smalleer than 32
280  // Send the input 12 bits: 7 MSB in CC Controller NUmber and the 5 LSB on CC Number+32
281  if( !typeSlider) SendData = (~PortValue)&0x0fff;
282  else SendData = (PortValue)&0x0fff;
283  MidiMerge.sendControlChange (PortCfg.ControllerNumber, (SendData>>5), PortCfg.MIDIChannel);
284  MidiMerge.sendControlChange (PortCfg.ControllerNumber+32, (SendData & 0x01f)<<2, PortCfg.MIDIChannel);
285  break;
286  case CONTROLCHANGE:
287  MidiMerge.sendControlChange (PortCfg.ControllerNumber, SendData, PortCfg.MIDIChannel);
288  break;
289  case PROGRAMCHANGE:
290  MidiMerge.sendProgramChange (SendData, PortCfg.MIDIChannel);
291  break;
292  case PITCHBEND:
293  MidiMerge.sendPitchBend (SendData, PortCfg.MIDIChannel);
294  break;
295  case PERCENT:
296  break;
297  case ANAGCLOCK:
298  break;
299  /* case ANAGSTARTSTOP:
300  if (IsHigh ()) {
301  if (GateStatus == 0) {
302  MidiMerge.sendRealTime (usbMIDI.Start);
303  GateStatus = 1;
304  }
305  } else if (GateStatus == 1) {
306  MidiMerge.sendRealTime (usbMIDI.Stop);
307  GateStatus = 0;
308  }
309  break;*/
310  case NOANFFUNC:
311  break;
312  case ANAGSTARTSTOP:
313  if (GateStatus == true)
314  MidiMerge.sendRealTime (usbMIDI.Start);
315  else
316  MidiMerge.sendRealTime (usbMIDI.Stop);
317  break;
318  case ANAGPAUSECONT:
319  if (GateStatus == true)
320  MidiMerge.sendRealTime (usbMIDI.Continue);
321  else
322  MidiMerge.sendRealTime (usbMIDI.Stop);
323  break;
324  case ANAGTRIGSTARTSTOP:
325  if (GateStatus == true) {
326  if (!MidiMerge.IsRunning)
327  MidiMerge.sendRealTime (usbMIDI.Start);
328  else
329  MidiMerge.sendRealTime (usbMIDI.Stop);
330  }
331  break;
332  case ANAGTRIGPAUSECONT:
333  if (GateStatus == true) {
334  if (!MidiMerge.IsRunning)
335  MidiMerge.sendRealTime (usbMIDI.Continue);
336  else
337  MidiMerge.sendRealTime (usbMIDI.Stop);
338  }
339  break;
340  case ANAGSTART:
341  if (GateStatus == true) MidiMerge.sendRealTime (usbMIDI.Start);
342  break;
343  case ANAGCONTINUE:
344  if (GateStatus == true) MidiMerge.sendRealTime (usbMIDI.Continue);
345  break;
346  case ANAGSTOP:
347  if (GateStatus == true) MidiMerge.sendRealTime (usbMIDI.Stop);
348  break;
349  case ANAGNRPN7bits:
350  MidiMerge.beginNrpn( (PortCfg.NRPNparMSB<<7)+PortCfg.NRPNparLSB, PortCfg.MIDIChannel);
351  MidiMerge.sendControlChange (6, SendData, PortCfg.MIDIChannel );
352  MidiMerge.endNrpn( PortCfg.MIDIChannel);
353  break;
354  case ANAGNRPN14bits:
355  MidiMerge.beginNrpn( (PortCfg.NRPNparMSB<<7)+PortCfg.NRPNparLSB, PortCfg.MIDIChannel);
356  MidiMerge.sendNrpnValue ( SendData, PortCfg.MIDIChannel );
357  MidiMerge.endNrpn( PortCfg.MIDIChannel);
358  break;
359  case ANAGCCLATCH:
360  case ANAGCCBUT:
361  if (GateStatus == true)
362  MidiMerge.sendControlChange (PortCfg.ControllerNumber, PortCfg.ControllerValue, PortCfg.MIDIChannel);
363  else
364  MidiMerge.sendControlChange (PortCfg.ControllerNumber, 0, PortCfg.MIDIChannel);
365  break;
366  case ANAGLATCH:
367  case ANAGTRIGGER:
368  if (GateStatus == true)
369  {
370  // Send new Note
371  MidiMerge.sendNoteOn( PortCfg.NoteToSend, MidiMerge.VelData[PortCfg.MIDIChannel-1], PortCfg.MIDIChannel);
372  MidiMerge.PitchData[PortCfg.MIDIChannel-1] = SendData;
373  }
374  else
375  {
376  MidiMerge.sendNoteOff( PortCfg.NoteToSend, 0, PortCfg.MIDIChannel);
377  MidiMerge.PitchData[PortCfg.MIDIChannel-1] = 0;
378  }
379  break;
380  }
381  // Store current data
382  LastSentMIDIData = SendData;
383  msecLastMIDISent = millis();
384 
385 }
