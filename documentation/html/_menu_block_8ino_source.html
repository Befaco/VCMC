179 boolean SelectCVConfig()
180 {
181  if (BankSelected == 8)
182  {
183  if (PortSelected < 2)
184  return true;
185  }
186  else
187  PortSelected = 2;
188 
189  #ifdef CVTHING
190  listAnag.disableItem(4);
191  #endif
192 
193  listAnag.disableItem(6);
194  listAnag.disableItem(7);
195  listAnag.disableItem(8);
196  listAnag.disableItem(9);
197  listAnag.disableItem(10);
198  listAnag.disableItem(11);
199  listAnag.disableItem(12);
200  // Show clock Div
201  if (CVControls[BankSelected].CVPort.PortCfg.MIDIfunction == ANAGCLOCK)
202  {
203  listAnag.enableItem(8);
204  }
205  // Show NRPN
206  if (CVControls[BankSelected].CVPort.PortCfg.MIDIfunction == ANAGNRPN14bits ||
207  CVControls[BankSelected].CVPort.PortCfg.MIDIfunction == ANAGNRPN7bits)
208  {
209  listAnag.enableItem(12);
210  listAnag.enableItem(10);
211  listAnag.enableItem(9);
212  }
213  if (CVControls[BankSelected].CVPort.PortCfg.MIDIfunction == PITCH)
214  {
215  listAnag.enableItem(11);
216  }
217  myMenu.ClearArea();
218  myMenu.setCurrentMenu(&listAnag);
219  return true;
220 }

267 boolean SelectNoteMode() {
268  CVControls[BankSelected].Config.Chanfunction = NOTEMODE;
269  // When NoteMode selected, CV use PITCHTRIG, Fader use Velocity and Gate is TRIGGER
270  CVControls[BankSelected].CVPort.PortCfg.SetMIDIFunc(PITCHTRIG);
271  CVControls[BankSelected].Slider.PortCfg.SetMIDIFunc(VELOCITY);
272  CVControls[BankSelected].GateBut.PortCfg.SetMIDIFunc(TRIGGER);
273 
274  return gotoMenuBanks();
275  /*myMenu.ClearArea();

277  return true;*/
278 }
