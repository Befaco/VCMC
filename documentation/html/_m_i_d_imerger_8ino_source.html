53 bool MIDImerger::poll (void) {
54  bool activity = false;
55  byte type = 0, channel, data1, data2;
56 
57  if (usbMIDI.read ()) {
58  // Thru on A has already pushed the input message to out A.
59  // Forward the message to out B as well.
60  // get the USB MIDI message, defined by these 5 numbers (except SysEX)
61  type = usbMIDI.getType ();
62  channel = usbMIDI.getChannel ();
63  data1 = usbMIDI.getData1 ();
64  data2 = usbMIDI.getData2 ();
65  // byte cable = usbMIDI.getCable();
66 
67  // forward this message the Serial MIDI OUT ports
68  if (type != usbMIDI.SystemExclusive) {
69  // Normal messages, first we must convert usbMIDI's type (an ordinary
70  // byte) to the MIDI library's special MidiType.
71  if( MIDIThru){
72  midi::MidiType mtype = (midi::MidiType)type;
73  midiJack->send (mtype, data1, data2, channel);
74  }
75  } else {
76  // SysEx messages are special. The message length is given in data1 & data2
77  unsigned int SysExLength = data1 + data2 * 256;
78  FlashAccess->ReceivedSysEx (SysExLength, usbMIDI.getSysExArray ());
79  if( MIDIThru){
80  midiJack->sendSysEx (SysExLength, usbMIDI.getSysExArray (), true);
81  }
82  }
83  activity = true;
84  }
85 
86  if (midiJack->read ()) {
87  // Thru on B has already pushed the input message to out B.
88  // Forward the message to out A as well.
89  // get a MIDI IN1 (Serial) message
90  type = midiJack->getType ();
91  channel = midiJack->getChannel ();
92  data1 = midiJack->getData1 ();
93  data2 = midiJack->getData2 ();
94 
95  // forward the message to USB MIDI virtual cable #0
96  if (type != midi::SystemExclusive) {
97  // Normal messages, simply give the data to the usbMIDI.send()
98  if( MIDIThru){
99  usbMIDI.send (type, data1, data2, channel, 0);
100  }
101  } else {
102  // SysEx messages are special. The message length is given in data1 & data2
103  unsigned int SysExLength = data1 + data2 * 256;
104  FlashAccess->ReceivedSysEx (SysExLength, midiJack->getSysExArray ());
105  if( MIDIThru){
106  usbMIDI.sendSysEx (SysExLength, midiJack->getSysExArray (), true, 0);
107  }
108  }
109  activity = true;
110  }
111 
112  if (FlashAccess->IsSysExRecComplete () && MidiMerge.tagProcessSysex == 0) {
113  if (MidiMerge.tagProcessSysex == 0) {
114  myMenu.ClearArea ();
115  myMenu.setCurrentMenu (&listLoadSysEx);
116  MidiMerge.tagProcessSysex = 1;
117  }
118  // SetMessageText( "SysEx Config Received!");
119  //&& myMenu.currentMenu!=&listLoadSysEx
120  }
121 
122  switch (type) {
123  case usbMIDI.Start:
124  case usbMIDI.Continue:
125  IsRunning = true;
126  SetMessageText ("Start/Cont.");
127  break;
128  case usbMIDI.Stop:
129  IsRunning = false;
130  SetMessageText ("Stop");
131  break;
132  }
133 
134  return activity;
135 }
