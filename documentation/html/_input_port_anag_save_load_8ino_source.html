102 int AnalogPort::parse (uint8_t type, uint8_t *buf, int buLen) {
103  int Initbuf = (int)buf;
104  int cont;
105  SysExDataStruct *pDataStr;
106  int16_t minMIDI, rMIDI;
107  int32_t minDAC, rDAC;
108 
109  if(typeSlider){
110  cont = sizeof(DATA_Fader) / sizeof(SysExDataStruct);
111  pDataStr = DATA_Fader;
112  }else{
113  cont = sizeof(DATA_Analog) / sizeof(SysExDataStruct);
114  pDataStr = DATA_Analog;
115  }
116  switch (type)
117  {
118  case DUMPGENERAL: // Dump directly buffer to class
119  for (int i = 1; i < cont;i++){
120  buf += parse(pDataStr[i].ParID, buf,
121  sizeTypes[pDataStr[i].Type]);
122  //D(Serial.printf("Digital Par %2d mem %d/%1d\n", pDataStr[i].ParID,
123  // buf, sizeTypes[pDataStr[i].Type]));
124  }
125  return (int)buf-Initbuf;
126  //LentoCopy = sizeof(PortCfg);
127  //memcpy(&PortCfg, buf, LentoCopy);
128  //break;
129  case PAR_MIDIChannel:
130  PortCfg.MIDIChannel = *buf;
131  return buLen;
132  case PAR_PortName:
133  PortCfg.textPort = *buf;
134  return buLen;
135  case PAR_NameSufix:
136  PortCfg.charSufix = *buf;
137  return buLen;
138  case PAR_DelayGate:
139  PortCfg.DelayGate = *((uint16_t*)buf);
140  return buLen;
141  case PAR_Function:
142  PortCfg.MIDIfunction = *buf;
143  return buLen;
144  case PAR_Range_minMIDI:
145  PortCfg.Ranges.getMIDI(minMIDI, rMIDI);
146  PortCfg.Ranges.SetMIDI(*((int16_t *)buf),rMIDI);
147  return buLen;
148  case PAR_Range_rangeMIDI:
149  PortCfg.Ranges.getMIDI(minMIDI, rMIDI);
150  PortCfg.Ranges.SetMIDI(minMIDI,*((int16_t *)buf));
151  return buLen;
152  case PAR_Range_minDAC:
153  PortCfg.Ranges.getDAC(minDAC, rDAC);
154  PortCfg.Ranges.SetDAC(*((int32_t *)buf),rDAC);
155  return buLen;
156  case PAR_Range_rangeDAC:
157  PortCfg.Ranges.getDAC(minDAC, rDAC);
158  PortCfg.Ranges.SetDAC(minDAC, *((int32_t *)buf));
159  return buLen;
160  case PAR_ClipLow:
161  PortCfg.ClipLow = *((int16_t*)buf);
162  return buLen;
163  case PAR_ClipHigh:
164  PortCfg.ClipHigh = *((int16_t*)buf);
165  return buLen;
166  case PAR_FunctionData:
167  return parseFunctionData( buf, buLen);
168  case PAR_Option1:
169  PortCfg.Options1 = *buf;
170  default:
171  return 0;
172  }
173  return 0;
174 }

177 int AnalogPort::parseFunctionData(uint8_t *buf, int buLen){
178 
179  if(buLen<2){
180  DP("Incorrect Length parsing parameter");
181  return 0;}
182 
183  switch(PortCfg.MIDIfunction){
184  case PITCH: // Dude mode
185  PortCfg.AutoOff = *((uint16_t *)buf);
186  return buLen;
187  case CC14BITS:
188  case CONTROLCHANGE:
189  PortCfg.ControllerNumber = *buf;
190  return buLen;
191  case ANAGCLOCK:
192  // get clock info from General config
193  PortCfg.ClockDivider = theApp.theGlobalCfg.ClockDivider;
194  PortCfg.ClockShift = theApp.theGlobalCfg.ClockShift;
195  return buLen;
196  case ANAGCCBUT:
197  case ANAGCCLATCH:
198  PortCfg.ControllerNumber = buf[0];
199  PortCfg.ControllerValue = buf[1];
200  return buLen;
201  case ANAGLATCH:
202  case ANAGTRIGGER:
203  PortCfg.NoteToSend = *buf;
204  return buLen;
205  case ANAGNRPN7bits:
206  case ANAGNRPN14bits:
207  PortCfg.NRPNparLSB = buf[0];
208  PortCfg.NRPNparMSB = buf[1];
209  return buLen;
210  case PITCHTRIG:
211  case VELOCITY:
212  case PROGRAMCHANGE:
213  case PITCHBEND:
214  case PERCENT:
215  case ANAGSTARTSTOP:
216  case ANAGPAUSECONT:
217  case ANAGTRIGSTARTSTOP:
218  case ANAGTRIGPAUSECONT:
219  case ANAGFREEVALUE:
220  case ANAGSTART:
221  case ANAGSTOP:
222  case ANAGCONTINUE:
223  case NOANFFUNC:
224  case PITCHLEVEL:
225  case PITCH8TRIG:
226  return buLen;
227  }
228  return 0;
229 }
