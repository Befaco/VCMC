143

146  return 0;}
147 
148  switch(PortCfg.MIDIfunction){

150  case LATCH:

152  return buLen;

154  case CCLATCH:

162  return buLen;
163  case GATESTARTSTOP:
164  case GATEPAUSECONT:
165  case GATETRIGSTARTSTOP:
166  case GATETRIGPAUSECONT:
167  case GATESTART:
168  case GATESTOP:
169  case GATECONTINUE:
170  case GATEPANIC:
