46 void VCMCApp::ProcessSysExCommand(VCMCSysExPacket *SysExPacket){
47  uint8_t bComm = SysExPacket->Slot; // Global commands are stored in the slot as 7 bits value
48  if( SysExPacket->Length <1) return;
49 
50  switch(bComm){
51  case SENDCURRENTCONFIG:
52  DP("Sending Current SysEx config");
53  FlashAcc.SaveCfgSysEx(-1);
54  break;
55  case RESETMODULE:
56  DP("Reset module");
57  ResetCPU();
58  break;
59  case FACTORYDEFAULT:
60  DP("Reset module to factory default");
61  FactoryReset();
62  break;
63  case SAVETOSLOT:
64  D(Serial.printf("Save to Slot %d.\n",SysExPacket->pData[0]));
65  FlashAcc.SaveCfg(SysExPacket->pData[0]);
66  break;
67  case LOADSLOT:
68  DP("Load Slot");
69  FlashAcc.LoadCfg(SysExPacket->pData[0]);
70  break;
71  default:
72  DP("SysEx General command not recognized");
73  }
74 }
