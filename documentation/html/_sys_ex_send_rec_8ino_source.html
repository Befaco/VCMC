118 bool SaveLoadClass::SaveCfgSysEx(byte *pSource, uint16_t uLengthSource)
119 {
120  uint8_t dumpMemory[uLengthSource * 8 / 7 + 8];
121  uint16_t posDump = 0;
122  union {
123  unsigned encLen = 0;
124  uint8_t bLen[2];
125  };
126  union {
127  unsigned long ulMicros;
128  uint8_t nIDMsg[4];
129  };
130  GenericSysExPacket SysExPacket;
131  uint8_t pCount = 0, csum = 0;
132  uint8_t MIDIencodedData[9];
133  unsigned encDataLen;
134  unsigned i;
135 
136  #ifdef PRINTDEBUG
137  Serial.print("Total Unencoded: ");
138  Serial.println(uLengthSource);
139  Serial.println("Unencoded Data: ");
140  for (i = 0; i < uLengthSource; i++)
141  {
142  Serial.print(pSource[i] >> 4, HEX);
143  Serial.print(pSource[i] & 0x0F, HEX);
144  Serial.print(" ");
145  }
146  Serial.println("<-End Data");
147  #endif
148  // Calculate checksum
149  csum = pSource[0];
150  for (i = 1; i < uLengthSource; i++)
151  csum ^= pSource[i];
152  csum &= 0x7F; // Strip to MIDI standard
153  #ifdef PRINTDEBUG
154  Serial.print("CheckSum: ");
155  Serial.println(csum);
156  #endif
157 
158  // encode to 7 bit MIDI leavinh two bytes for message length
159  encLen = midi::encodeSysEx((const byte *)pSource, dumpMemory + 2, uLengthSource);
160  dumpMemory[0] = csum;
161  encLen += 2; // Add checksum byte
162  #ifdef PRINTDEBUG
163  Serial.print("Encoded Length: ");
164  Serial.println(encLen);
165  #endif
166  // Calculate number of messages
167  pCount = encLen / SysExpacketDataLength;
168  pCount += (pCount * SysExpacketDataLength < encLen) ? 1 : 0;
169  dumpMemory[1] = pCount;
170  #ifdef PRINTDEBUG
171  Serial.print("# Messages: ");
172  Serial.println(pCount);
173  #endif
174  pCount = 0;
175 
176  // Store data length to beginning of message (Not including these two bytes nor trailing
177  // checksum byte) COnvert to 7 bit length
178  // encDataLen = midi::encodeSysEx((const byte*)&encLen, MIDIencodedData, 2);
179  // memcpy( dumpMemory+encLen, MIDIencodedData, encDataLen);
180 
181  // Create a message ID using millis function (useful as timing also?)
182  ulMicros = millis();
183  // Mark first 16 bits as CFGDATATAG to mark as a config message
184  ulMicros = (ulMicros & 0x0000FFFF) + (((unsigned long)CFGDATATAG) << 16);
185  encDataLen = midi::encodeSysEx((const byte *)nIDMsg, MIDIencodedData, 4);
186  memcpy(SysExPacket.ID, MIDIencodedData, encDataLen);
187 
188  // Send SysEx messages with length + encoded data + checksum.
189  // Each SysEx contains a maximum of SysExpacketDataLength bytes
190  unsigned long pausMil = millis();
191  while (posDump < encLen - SysExpacketDataLength)
192  {
193  if (pausMil + 30 < millis())
194  { // Wait between messages sent out
195  #ifdef PRINTDEBUG
196  Serial.print("Send SysEx Packet #: ");
197  Serial.print(pCount);
198  Serial.print(" / ");
199  Serial.print(posDump);
200  Serial.println(" data bytes");
201  #endif
202  SysExPacket.pCounter = pCount;
203  pCount++;
204  memcpy(SysExPacket.pData, dumpMemory + posDump, SysExpacketDataLength);
205  posDump += SysExpacketDataLength;
206  usbMIDI.sendSysEx(sizeof(SysExPacket), (const uint8_t *)&SysExPacket, true);
207  pausMil = millis();
208  }
209  }
210  if (posDump >= encLen - SysExpacketDataLength)
211  {
212  #ifdef PRINTDEBUG
213  Serial.print("Last Send SysEx Packet #: ");
214  Serial.print(pCount);
215  Serial.print(" / ");
216  Serial.print(posDump);
217  Serial.println(" data bytes");
218  #endif
219  SysExPacket.pCounter = pCount;
220  memcpy(SysExPacket.pData, dumpMemory + posDump, encLen - posDump);
221  SysExPacket.pData[encLen - posDump] = 0xF7; // Mark the EOX
222  usbMIDI.sendSysEx(sizeof(SysExPacket), (const uint8_t *)&SysExPacket, true);
223  }
224  return true;
225 }

235 bool SaveLoadClass::ReceivedSysEx(unsigned SysExLength, const byte *RecArray)
236 {
237  GenericSysExPacket *SysExPacket = (GenericSysExPacket *)RecArray;
238  unsigned long RecPacketID;
239 
240  #ifdef PRINTDEBUG
241  Serial.print("SysEx Device: ");
242  Serial.println(SysExPacket->Dev, HEX);
243  #endif
244  if (SysExPacket->Dev != 0x7D){
245  #ifdef PRINTDEBUG
246  Serial.print("Incorrect SysEx Device: ");
247  Serial.println(SysExPacket->Dev, HEX);
248  #endif
249  return false; // Not for us!
250  }
251  #ifdef PRINTDEBUG
252  Serial.print("SysEx Received length: ");
253  Serial.println(SysExLength);
254  #endif
255  if (SysExLength > 64 || SysExLength < 10)
256  {
257  D(Serial.println("SysEx size not valid "));
258  SetMessageText("SysEx size not valid");
259  return false; // Not expected size SysEx
260  }
261 
262  if(SysExPacket->ID[0]==0x32)
263  {
264  // Single message SysEx received
265  return ReceivedSingleSysEx(SysExLength, RecArray);
266  }
267 
268  // Decode message ID
269  midi::decodeSysEx((const byte *)SysExPacket->ID, (byte *)&RecPacketID, 5);
270  #ifdef PRINTDEBUG
271  Serial.print("SysEx Received ID: ");
272  Serial.println(RecPacketID, HEX);
273  #endif
274  //if ((RecPacketID >> 28) != 6) {
275  if ((RecPacketID >> 16) != CFGDATATAG)
276  {
277  D(Serial.println("SysEx Received ID not valid "));
278  //SetMessageText ("SysEx ID not valid");
279  myMenu.setupPopup("SysEx ID not valid", 1000, 10, 20, 0);
280  return false; // Not a config message, should begin with 0x60000000
281  }
282  RecPacketID = RecPacketID & 0x0FFFFFFF; // Remove check bits
283 
284  // Check if message is already received
285  if (PacketsRec[SysExPacket->pCounter] && ulPacketID == RecPacketID)
286  {
287  #ifdef PRINTDEBUG
288  Serial.print("SysEx Packet #: ");
289  Serial.print(SysExPacket->pCounter);
290  Serial.println(" already received");
291  #endif
292  return false;
293  }
294  char outBuf[80];
295  strncpy(outBuf, "SysEx Rec:", 11);
296  padvalue(SysExPacket->pCounter, outBuf + 10, 2);
297  outBuf[2 + 10] = '/';
298  padvalue(nPacketsRec, outBuf + 3 + 10, 2);
299  SetMessageText(outBuf);
300  #ifdef PRINTDEBUG
301  Serial.print("SysEx Packet #: ");
302  Serial.print(SysExPacket->pCounter);
303  Serial.print(" / ");
304  Serial.println(nPacketsRec);
305  #endif
306 
307  // If this is the first message or a new message ID, initialise variables
308  if (!nPacketsRec || ulPacketID != RecPacketID)
309  {
310  ResetReception();
311  ulPacketID = RecPacketID;
312  nPacketsRec = 1;
313  MidiMerge.tagProcessSysex = 0;
314  }
315  else
316  nPacketsRec++;
317  PacketsRec[SysExPacket->pCounter] = 1;
318 
319  if (SysExPacket->pCounter == 0)
320  { // First message received
321  msgCsum = SysExPacket->pData[0]; // Store Checksum
322  nPacketsTotal = SysExPacket->pData[1]; // Store # packages
323  #ifdef PRINTDEBUG
324  Serial.print("SysEx CheckSum: ");
325  Serial.println(msgCsum);
326  Serial.print("SysEx Msgs to receive: ");
327  Serial.println(nPacketsTotal);
328  #endif
329  }
330  // Store data
331  #ifdef PRINTDEBUG
332  Serial.println("SysEx Data: ");
333  for (unsigned i = 0; i < SysExLength - 9; i++)
334  {
335  Serial.print(SysExPacket->pData[i] >> 4, HEX);
336  Serial.print(SysExPacket->pData[i] & 0x0F, HEX);
337  }
338  Serial.println("<-End Data");
339  #endif
340  memcpy(rcvSysEx + SysExPacket->pCounter * SysExpacketDataLength, SysExPacket->pData, SysExLength - 9);
341  nBytesRec += SysExLength - 9;
342  #ifdef PRINTDEBUG
343  Serial.print("SysEx Total Received: ");
344  Serial.println(nBytesRec);
345  #endif
346 
347  if (IsSysExRecComplete())
348  {
349  SetMessageText("SysEx Config Received!");
350  }
351  // Process finished message
352  #ifdef PRINTDEBUG
353  if (!IsSysExRecComplete())
354  return true;
355  Serial.println("Encoded Data: ");
356  for (unsigned i = 0; i < nBytesRec; i++)
357  {
358  Serial.print(rcvSysEx[i] >> 4, HEX);
359  Serial.print(rcvSysEx[i] & 0x0F, HEX);
360  }
361  Serial.println("<-End Data");
362  byte csum = calcCSum(); // Decode data and get Checksum
363  Serial.print("SysEx Total Decoded: ");
364  Serial.println(nDecoded);
365  Serial.println("Decoded Data: ");
366  for (unsigned i = 0; i < nDecoded; i++)
367  {
368  Serial.print(rcvData[i] >> 4, HEX);
369  Serial.print(rcvData[i] & 0x0F, HEX);
370  Serial.print(" ");
371  }
372  Serial.println("<-End Data");
373  Serial.print("SysEx CheckSum: ");
374  Serial.println(msgCsum);
375  Serial.print("SysEx Data Checksum: ");
376  Serial.println(csum);
377  #endif
378 
379  return true;
380 }
