71 bool DigitalPort::ReadPort (long &NewData) {
72  if (PortNumber == 0) return false;
73  if (PortCfg.MIDIfunction == NODIGFUNC) return false; // No Function, do nothing
74 
75  // Process MIDI clock
76  if (PortCfg.MIDIfunction == GATECLOCK && IntervalMIDIClock && ClockCounter) {
77  ProcessClock();
78  }
79 
80  // read the state of the switch into a local variable:
81  unsigned reading = digitalRead (PortNumber);
82  int newGate = 2;
83 
84  if (debounceGate) { // Debounce function if selected
85  // check to see if you just pressed the button
86  // (i.e. the input went from LOW to HIGH), and you've waited
87  // long enough since the last press to ignore any noise:
88  // If the switch changed, due to noise or pressing:
89  if (reading != lastButtonState) {
90  // reset the debouncing timer
91  lastDebounceTime = millis ();
92  }
93 
94  if ((millis () - lastDebounceTime) > PortCfg.DelayGate) {
95  // whatever the reading is at, it's been there for longer
96  // than the debounce delay, so take it as the actual current state:
97  newGate = reading;
98  }
99  lastButtonState = reading;
100  } else
101  newGate = reading;
102 
103  if (newGate == 2) return false; // No lecture (debouncing)
104  // PortValue = reading;
105  if (newGate != PortValue) { // Gate changed
106  PortValue = newGate;
107  #if GATEPULLUP
108  newGate = (newGate == 0) ? 1 : 0; // !newGate;
109  #endif
110  // Latch management: switch gate with edge up
111  if (PortCfg.MIDIfunction == LATCH || PortCfg.MIDIfunction == CCLATCH) {
112  if (newGate) {
113  GateStatus = !GateStatus;
114  } else
115  return false;
116  } else
117  GateStatus = newGate;
118  NewData = GateStatus;
119  // Lit LED based on gate status
120  digitalWrite (LedPort, GateStatus);
121 
122  // Process Clock
123  if (PortCfg.MIDIfunction == GATECLOCK)
124  if( GateStatus) ClockReceived();
125  return true;
126  }
127 
128  return false;
129 }

164 void DigitalPort::SendMIDI (int MidiData, bool GateStat) {
165  msecLastMIDISent = millis();
166 
167  switch (PortCfg.MIDIfunction) {
168  case TRIGGER:
169  case LATCH:
170  break;
171  case CCLATCH:
172  case CCBUT:
173  if (GateStatus == true)
174  MidiMerge.sendControlChange (PortCfg.ControllerNumber, PortCfg.ControllerValue, PortCfg.MIDIChannel);
175  else
176  MidiMerge.sendControlChange (PortCfg.ControllerNumber, 0, PortCfg.MIDIChannel);
177  break;
178  case GATECLOCK:
179  break;
180  case GATESTARTSTOP:
181  if (GateStatus == true)
182  MidiMerge.sendRealTime (usbMIDI.Start);
183  else
184  MidiMerge.sendRealTime (usbMIDI.Stop);
185  break;
186  case GATEPAUSECONT:
187  if (GateStatus == true)
188  MidiMerge.sendRealTime (usbMIDI.Continue);
189  else
190  MidiMerge.sendRealTime (usbMIDI.Stop);
191  break;
192  case GATETRIGSTARTSTOP:
193  if (GateStatus == true) {
194  if (!MidiMerge.IsRunning)
195  MidiMerge.sendRealTime (usbMIDI.Start);
196  else
197  MidiMerge.sendRealTime (usbMIDI.Stop);
198  }
199  break;
200  case GATETRIGPAUSECONT:
201  if (GateStatus == true) {
202  if (!MidiMerge.IsRunning)
203  MidiMerge.sendRealTime (usbMIDI.Continue);
204  else
205  MidiMerge.sendRealTime (usbMIDI.Stop);
206  }
207  break;
208  case GATESTART:
209  if (GateStatus == true) MidiMerge.sendRealTime (usbMIDI.Start);
210  break;
211  case GATECONTINUE:
212  if (GateStatus == true) MidiMerge.sendRealTime (usbMIDI.Continue);
213  break;
214  case GATESTOP:
215  if (GateStatus == true) MidiMerge.sendRealTime (usbMIDI.Stop);
216  break;
217  case GATEPANIC:
218  if (GateStatus == true) MidiMerge.Panic();
219  case NODIGFUNC:
220  break;
221  }
222 }
