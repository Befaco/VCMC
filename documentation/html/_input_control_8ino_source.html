174 void InputControl::OnDataChange (void) {
175  int SendData = DataCalc ();
176  bool GateStat = GetGateState ();
177 
178  // Blink LED
179  // GateBut.setBlink(100, 100, 1);
180 
181  // Gate change management
182  if (Gatechanged) {
183  if (GateBut.PortCfg.MIDIfunction == TRIGGER || GateBut.PortCfg.MIDIfunction == LATCH) { // In TRIGGER or LATCH send NoteOn/NoteOff with every change on inputs
184  if (GateStat == true) { // Send Note On
185  // Independ channels
186  boolean played = false;
187  if (Config.Chanfunction != INDEP) {
188  // If SUM or MULTIP send result through CV channel if it has the trigger option
189  if (CVPort.PortCfg.MIDIfunction == TRIGGER) {
190  byte datatosend = CVPort.TrimValue (SendData);
191  if( CVPort.LastSentMIDIData>0){ // sendNoteOff previous note
192  MidiMerge.sendNoteOff (GateBut.MIDIData, 0, CVPort.PortCfg.MIDIChannel);
193  CVPort.LastSentMIDIData=-999;}
194  CVPort.LastSentMIDIData = datatosend;
195  MidiMerge.sendNoteOn (datatosend, MidiMerge.VelData[CVPort.PortCfg.MIDIChannel - 1],
196  CVPort.PortCfg.MIDIChannel);
197  played = true;
198  // GateBut.setBlink(100, 100, 1);
199  }
200  } else {
201 
202  if (CVPort.PortCfg.MIDIfunction == PITCHTRIG || CVPort.PortCfg.MIDIfunction == PITCH8TRIG) { // CV triggered notes
203  byte datatosend = CVPort.TrimValue (CVPort.MIDIData);
204  if( CVPort.LastSentMIDIData>0){ // sendNoteOff previous note
205  MidiMerge.sendNoteOff (CVPort.LastSentMIDIData, 0, CVPort.PortCfg.MIDIChannel);
206  CVPort.LastSentMIDIData=-999;}
207  CVPort.LastSentMIDIData= datatosend; // Store Note in CVPort at gate activation
208  MidiMerge.sendNoteOn (datatosend, MidiMerge.VelData[CVPort.PortCfg.MIDIChannel - 1],
209  CVPort.PortCfg.MIDIChannel);
210  played = true;
211  }
212  if (Slider.PortCfg.MIDIfunction == PITCHTRIG) { // Fader triggered notes
213  byte datatosend = Slider.TrimValue (Slider.MIDIData);
214  if( Slider.LastSentMIDIData>0){ // sendNoteOff previous note
215  MidiMerge.sendNoteOff (Slider.LastSentMIDIData, 0, Slider.PortCfg.MIDIChannel);
216  Slider.LastSentMIDIData=-999;}
217  Slider.LastSentMIDIData = datatosend; // Store Note in Slider at gate activation
218  MidiMerge.sendNoteOn (datatosend, MidiMerge.VelData[Slider.PortCfg.MIDIChannel - 1],
219  Slider.PortCfg.MIDIChannel);
220  played = true;
221  }
222  }
223  if(!played){ // Gate play notes
224  byte datatosend = GateBut.PortCfg.NoteToSend;
225  if( GateBut.MIDIData){ // sendNoteOff previous note
226  MidiMerge.sendNoteOff (GateBut.MIDIData, 0, GateBut.PortCfg.MIDIChannel);
227  GateBut.MIDIData=0;}
228  GateBut.MIDIData = datatosend; // Store Note in CVPort at gate activation
229  MidiMerge.sendNoteOn (datatosend, MidiMerge.VelData[GateBut.PortCfg.MIDIChannel - 1], GateBut.PortCfg.MIDIChannel);
230  }
231 
232  }
233  else { // Send Note Off
234  boolean played = false;
235  // If SUM or MULTIP send result through CV channel if it has the trigger option
236  if (Config.Chanfunction != INDEP) {
237  if (CVPort.PortCfg.MIDIfunction == TRIGGER){
238  MidiMerge.sendNoteOff (CVPort.LastSentMIDIData, 0, CVPort.PortCfg.MIDIChannel);
239  CVPort.LastSentMIDIData=0;
240  played = true;
241  }
242  }
243  // Independ channels
244  else {
245  if (CVPort.PortCfg.MIDIfunction == PITCHTRIG || CVPort.PortCfg.MIDIfunction == PITCH8TRIG){
246  played = true;
247  MidiMerge.sendNoteOff (CVPort.LastSentMIDIData, 0, CVPort.PortCfg.MIDIChannel);
248  CVPort.LastSentMIDIData=-999;}
249  if (Slider.PortCfg.MIDIfunction == PITCHTRIG){
250  played = true;
251  MidiMerge.sendNoteOff (Slider.LastSentMIDIData, 0, Slider.PortCfg.MIDIChannel);
252  Slider.LastSentMIDIData=-999;}
253  }
254  if(!played) {
255  MidiMerge.sendNoteOff (GateBut.MIDIData, 0, GateBut.PortCfg.MIDIChannel);
256  GateBut.MIDIData=0;}
257  }
258  }
259  else if( ControlNumber== 7 && GateBut.PortCfg.MIDIfunction == GATE8TRIG ){ // Trigger all CVs with
260  for (int i = 0; i < 8;i++){
261  if(CVControls[i].CVPort.PortCfg.MIDIfunction== PITCH8TRIG){
262  uint8_t oldFn = CVControls[i].GateBut.PortCfg.MIDIfunction;
263  CVControls[i].GateBut.PortCfg.MIDIfunction = TRIGGER;
264  CVControls[i].GateBut.GateStatus = GateBut.GateStatus;
265  CVControls[i].Gatechanged = true;
266  CVControls[i].OnDataChange();
267  CVControls[i].Gatechanged = false;
268  CVControls[i].GateBut.PortCfg.MIDIfunction = oldFn;
269  }
270  }
271  } else
272  GateBut.SendMIDI ();
273  }
274  // Apply data calculation based on selected function
275  // In case of operations performed, send calculated data using CVport
276  if (CVchanged || Slidchanged) {
277  switch (Config.Chanfunction) {
278  case NOTEMODE:
279  break;
280  case INDEP:
281  // Do not send Notes CV changes on NOTEMODE
282  if (CVchanged) {
283  #ifdef CVTHING
284  if (CVPort.PortCfg.MIDIfunction == PITCHLEVEL) { // send Trigger to paired CV
285  if( ControlNumber<1)
286  break;
287  if( ControlNumber == 7 && CVControls[6].CVPort.PortCfg.MIDIfunction!=PITCHTRIG){ // Port 8 as trigger
288  for (int i = 0; i < 7;i++){
289  if(CVControls[i].CVPort.PortCfg.MIDIfunction== PITCH8TRIG){
290  CVControls[i].GateBut.GateStatus = CVPort.IsHigh();
291  CVControls[i].Gatechanged = true;
292  CVControls[i].OnDataChange();
293  CVControls[i].Gatechanged = false;
294  }
295  }
296  } else
297  {
298  CVControls[ControlNumber - 1].GateBut.GateStatus = CVPort.IsHigh();
299  CVControls[ControlNumber - 1].Gatechanged = true;
300  CVControls[ControlNumber - 1].OnDataChange();
301  CVControls[ControlNumber - 1].Gatechanged = false;
302  }
303 
304  return;
305  }
306  //PITCH8TRIG
307  #endif
308  CVPort.SendMIDI (-9999, GateStat);
309  }
310  if (Slidchanged) {
311  Slider.SendMIDI (-9999, GateStat);
312  }
313  break;
314  case SUM:
315  case MULTIPL:
316  CVPort.SendMIDI (SendData, GateStat);
317  // GateBut.setBlink(100, 100, 1);
318  }
319  }
320 }

329 int InputControl::DataCalc (void) {
330  int outdata = -9999;
331 
332  switch (Config.Chanfunction) {
333  case NOTEMODE: //*** Review
334  case INDEP:
335  return -9999;
336  case SUM:
337  outdata = (CVPort.MIDIData + Slider.MIDIData);
338  break;
339  case MULTIPL:
340  // Overflow due to result bigger than int is not managed
341  // When the slider is set to percent, scale value down by 100
342  if (Slider.PortCfg.MIDIfunction == PERCENT)
343  outdata = ((long)CVPort.MIDIData * Slider.MIDIData) / 100;
344  else
345  outdata = ((long)CVPort.MIDIData * Slider.MIDIData);
346  }
347  byte bBip = CVPort.PortCfg.getInputRange();
348  CVPort.PortCfg.setInputRange(NOOFFSET);
349  outdata = CVPort.TrimValue(outdata);
350  CVPort.PortCfg.setInputRange(bBip);
351 
352  return outdata; // CVPort.TrimValue(outdata);
353 }
