148 void VCMCApp::initdisplay(void)
149 {
150  // by default, we'll generate the high voltage from the 3.3v line internally! (neat!)
151  theOLED->begin(SSD1306_SWITCHCAPVCC);
152  SPI.begin();
153  // Moce SCK from pin 13 to pin 14
154  SPI.setSCK(14);
155  // Set SPI speed
156  SPI.setClockDivider(SPI_CLOCK_DIV2); // 2.25 MHz
157 
158  digitalWrite(OLED_RESET, HIGH);
159  delay(10);
160  digitalWrite(OLED_RESET, LOW);
161  delay(100);
162  digitalWrite(OLED_RESET, HIGH);
163  delay(100);
164 
165  digitalWrite(OLED_DC, LOW);
166  digitalWrite(OLED_CS, LOW);
167 
168  SPI.transfer(0xAE); // Display off
169  SPI.transfer(0xD5); // Set display clock divider
170  SPI.transfer(0x80);
171  SPI.transfer(0xA8); // Set multiplex
172  SPI.transfer(0x3F);
173  SPI.transfer(0xD3); // Set display offset
174  SPI.transfer(SCREEN_OFFSET);
175  SPI.transfer(0x40); // Set start line to zero
176  SPI.transfer(0x8D); // Set charge pump
177  SPI.transfer(0x14);
178  SPI.transfer(0x20); // Set memory mode
179  SPI.transfer(0x00);
180  SPI.transfer(0xA0 | 0x1); // Set segment remapping
181  SPI.transfer(0xC8); // Set command Scan decode
182  SPI.transfer(0xDA); // Set Comm pins
183  SPI.transfer(0x12);
184  SPI.transfer(0x81); // Set contrast
185  SPI.transfer(0xCF);
186  SPI.transfer(0xd9); // Set precharge
187  SPI.transfer(0xF1);
188  SPI.transfer(0xDB); // Set Vcom detect
189  SPI.transfer(0x30);//40);
190  SPI.transfer(0xA4); // Allow display resume
191  SPI.transfer(0xA6); // Set normal display
192  SPI.transfer(0xAF); // Display On
193 
194  //theOLED->setRotation(2); // Rotate the screen 180 degrees
195  // Clear the buffer.
196  theOLED->clearDisplay();
197  theOLED->setTextWrap(false);
198  theOLED->setTextColor(WHITE, BLACK);
199 }
